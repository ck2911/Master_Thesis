\chapter*{List of Symbols}
\addcontentsline{toc}{chapter}{List of Symbols}

\begin{description}[leftmargin=4.4cm,style=nextline]
  \item[\(\alpha,\beta,\gamma\)] Generic scalar coefficients.
  \item[\(i_t\)] Monetary-policy stance or policy instrument at time \(t\).
  \item[\(x_t\)] Generic macro-financial outcome at time \(t\).
  \item[\(u_t\)] Reduced-form innovation or residual.
  \item[\(\ell_t\)] Lending quantity.
  \item[\(s_t\)] Lending spread.
  \item[\(b_t\)] Bank balance-sheet conditions.
  \item[\(\vect{y}_t\)] Vector of macro-financial variables at time \(t\).
  \item[\(\vect{c}\)] Vector of intercept terms.
  \item[\(p\)] Lag order.
  \item[\(\mat{A}_j\)] Reduced-form autoregressive coefficient matrix for lag \(j\).
  \item[\(\vect{u}_t\)] Vector of reduced-form VAR residuals.
  \item[\(\mat{\Sigma}_u\)] Covariance matrix of reduced-form residuals.
  \item[\(\mat{B}_0,\mat{B}_j\)] Structural contemporaneous and lag coefficient matrices.
  \item[\(\vect{\shock}_t\)] Vector of structural shocks.
  \item[\(\policyshock_t\)] Monetary-policy structural shock.
  \item[\(\mat{I}\)] Identity matrix.
  \item[\(\IRF_i(h)\)] Impulse response of variable \(i\) at horizon \(h\).
  \item[\(\vect{e}_i,\vect{e}_{mp}\)] Selection vectors for variable \(i\) and the monetary-policy shock.
  \item[\(\mat{\Phi}_h\)] Moving-average coefficient matrix at horizon \(h\).
  \item[\(\surprise_t\)] High-frequency monetary-policy surprise.
  \item[\(\surprise_m^{target}\)] Monthly aggregate ECB target surprise for month \(m\).
  \item[\(\omega_e\)] Aggregation weight assigned to policy event \(e\).
  \item[\(\alpha_{i,h},\beta_{i,h}\)] Local-projection intercept and shock-response coefficient for variable \(i\) at horizon \(h\).
  \item[\(\vect{X}_{t-1}\)] Predetermined controls in the local-projection specification.
  \item[\(\vect{\Gamma}_{i,h}\)] Local-projection control coefficient vector.
  \item[\(\widehat{\sigma}_{\surprise}\)] Estimated standard deviation of the monetary-policy surprise.
  \item[\(\FEVD_i(H)\)] Forecast-error variance decomposition share for variable \(i\) through horizon \(H\).
\end{description}
