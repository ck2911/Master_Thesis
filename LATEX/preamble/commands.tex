% Figure import helper. The source note records the canonical artifact path.
\newcommand{\thesisfigure}[4][0.88\textwidth]{%
  \begin{figure}[htbp]
    \centering
    \IfFileExists{\figpath/#2}{%
      \includegraphics[width=#1]{\figpath/#2}%
    }{%
      \fbox{\begin{minipage}[c][0.25\textheight][c]{0.86\textwidth}
        \centering
        Missing canonical figure artifact:\par
        \nolinkurl{results/final/figures/#2}
      \end{minipage}}%
    }%
    \caption{#3}
    \label{#4}
    \vspace{0.25em}
    {\footnotesize Source: \nolinkurl{results/final/figures/#2}.}
  \end{figure}
}

% Table artifact helper. This keeps CSV files as the source of truth while
% registering the thesis-facing table in the list of tables.
\newcommand{\artifacttable}[4]{%

    \label{#2}
    \begin{tabularx}{\textwidth}{@{}p{0.28\textwidth}X@{}}
      \toprule
      Canonical artifact & \nolinkurl{#3} \\
      Thesis use & #4 \\
      \bottomrule
    \end{tabularx}
  \end{table}
}
